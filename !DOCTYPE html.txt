        function showForm(service) {
            const formContainer = document.getElementById('form-container');
            formContainer.style.display = 'block';

            const serviceName = document.getElementById('service-name');
            serviceName.innerText = service.name;

            const formFields = document.getElementById('form-fields');
            formFields.innerHTML = ''; 

            service.fields.forEach(field => {
                const fieldContainer = document.createElement('div');
                fieldContainer.innerHTML = `
                    <label for="${field.id}">${field.label}:</label><br>
                    <input type="${field.type}" id="${field.id}" name="${field.name}" required><br><br>
                `;
                formFields.appendChild(fieldContainer);
            });
        }

        function closeForm() {
            const formContainer = document.getElementById('form-container');
            formContainer.style.display = 'none';
        }

        function displayMessage(event) {
            event.preventDefault();
            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;
            const message = document.getElementById('message').value;
            const submittedMessage = document.getElementById('submitted-message');
